		
	#region "IEnumerable Members"
        
    %version%
    Private Function GetEnumerator1() As IEnumerator Implements IEnumerable.GetEnumerator

        Dim enumProxy As Object = Invoker.%Call%(Me, "_NewEnum")
        Dim enumerator As COMObject = New COMObject(Me, enumProxy, True)
        Invoker.Method(enumerator, "Reset", Nothing)
        Dim isMoveNextTrue As Boolean = Invoker.MethodReturn(enumerator, "MoveNext", Nothing)

        Do While (True = isMoveNextTrue)

            Dim itemProxy As Object = Invoker.PropertyGet(enumerator, "Current", Nothing)
            Dim returnValue As %Type% = itemProxy
            isMoveNextTrue = Invoker.MethodReturn(enumerator, "MoveNext", Nothing)
            Return returnValue

        Loop

        Return Nothing

    End Function

    #End Region
