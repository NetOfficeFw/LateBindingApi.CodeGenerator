   
        #region "IEnumerable Members"
		
        %version%
		<CustomEnumerator> _
		Private Function GetEnumerator1() As IEnumerator Implements IEnumerable.GetEnumerator
		
			Dim count As Integer = Me.Count
			Dim enumeratorObjects(count) As %Type%
	
	        For i As Integer = 0 To count-1
				enumeratorObjects(i) = Me(i+1)
		    Next
			
			For Each item as %Type% in enumeratorObjects
				return item
			Next
			
			Return Nothing	
        
		End Function

        #End Region
