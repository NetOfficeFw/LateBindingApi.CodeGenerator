        <EditorBrowsable(EditorBrowsableState.Never), Browsable(false)> _
        Public ReadOnly Property HasEventRecipients As Boolean
			
			Get
						
				If LateBindingApi.Core.Utils.IsNothing(_thisType)
					_thisType = Me.GetType()
				End If
					
				For Each item As NetRuntimeSystem.Reflection.EventInfo In _thisType.GetEvents()

					Dim eventDelegate As MulticastDelegate = _thisType.GetType().GetField(item.Name, NetRuntimeSystem.Reflection.BindingFlags.NonPublic Or NetRuntimeSystem.Reflection.BindingFlags.Instance).GetValue(Me)
					
					If LateBindingApi.Core.Utils.IsNothing(eventDelegate) And eventDelegate.GetInvocationList().Length > 0
						return false
					End If		
					
                Next
                				
				return true
			
			End Get
			
        End Property
        
		<EditorBrowsable(EditorBrowsableState.Never), Browsable(false)> _
        public Function GetEventRecipients(eventName As String) As Delegate() 
        
			If LateBindingApi.Core.Utils.IsNothing(_thisType)
					_thisType = Me.GetType()
				End If
             
            Dim eventDelegate As MulticastDelegate = _thisType.GetField("_" + eventName + "Event", NetRuntimeSystem.Reflection.BindingFlags.Instance Or NetRuntimeSystem.Reflection.BindingFlags.NonPublic).GetValue(Me)

			If Not LateBindingApi.Core.Utils.IsNothing(eventDelegate)
            
                Dim delegates() As Delegate = eventDelegate.GetInvocationList()
                return delegates
            
            Else
                return new Delegate(0)
            End If
                
        End Function
